--// UTILS
local function Create(class, props)
    local obj = Instance.new(class)
    for k, v in pairs(props or {}) do
        obj[k] = v
    end
    return obj
end

local function Tween(obj, props, duration, easing, direction)
    duration = duration or CONFIG.AnimSpeed
    easing = easing or Enum.EasingStyle.Quart
    direction = direction or Enum.EasingDirection.Out
    local tween = TweenService:Create(obj, TweenInfo.new(duration, easing, direction), props)
    tween:Play()
    return tween
end

--// ANTI-CHEAT BYPASS
local function SetupBypass()
    local mt = getrawmetatable(game)
    setreadonly(mt, false)
    local oldNamecall = mt.__namecall
    local oldIndex = mt.__index
    
    mt.__namecall = newcclosure(function(self, ...)
        local method = getnamecallmethod()
        if method == "Kick" and self == LocalPlayer then
            return
        end
        if method == "Destroy" and (self == LocalPlayer or self == LocalPlayer.Character) then
            return
        end
        return oldNamecall(self, ...)
    end)
    
    mt.__index = newcclosure(function(self, key)
        if self:IsA("Humanoid") then
            if key == "WalkSpeed" then return 16 end
            if key == "JumpPower" then return 50 end
        end
        return oldIndex(self, key)
    end)
    
    setreadonly(mt, true)
end

pcall(SetupBypass)

--// ROLE DETECTION
local RoleSystem = {
    CurrentRoles = {},
    Murderer = nil,
    Sheriff = nil,
    Hero = nil
}

function RoleSystem.DetectRoles()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local backpack = player:FindFirstChild("Backpack")
            local char = player.Character
            local hasKnife = false
            local hasGun = false
            
            if backpack then
                for _, tool in pairs(backpack:GetChildren()) do
                    if tool.Name:lower():find("knife") or tool.Name:lower():find("blade") then
                        hasKnife = true
                    end
                    if tool.Name:lower():find("gun") or tool.Name:lower():find("revolver") then
                        hasGun = true
                    end
                end
            end
            
            for _, tool in pairs(char:GetChildren()) do
                if tool:IsA("Tool") then
                    if tool.Name:lower():find("knife") or tool.Name:lower():find("blade") then
                        hasKnife = true
                    end
                    if tool.Name:lower():find("gun") or tool.Name:lower():find("revolver") then
                        hasGun = true
                    end
                end
            end
            
            if hasKnife then
                RoleSystem.Murderer = player
                RoleSystem.CurrentRoles[player] = "Murderer"
            elseif hasGun then
                if not RoleSystem.Sheriff then
                    RoleSystem.Sheriff = player
                    RoleSystem.CurrentRoles[player] = "Sheriff"
                else
                    RoleSystem.Hero = player
                    RoleSystem.CurrentRoles[player] = "Hero"
                end
            else
                RoleSystem.CurrentRoles[player] = "Innocent"
            end
        end
    end
end

--// ESP SYSTEM
local ESPSystem = {
    Active = false,
    ESPObjects = {}
}

function ESPSystem.CreateESP(player, role)
    if player == LocalPlayer then return end
    if not player.Character then return end
    
    local char = player.Character
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    ESPSystem.RemoveESP(player)
    
    local color
    if role == "Murderer" then color = CONFIG.MurdererColor
    elseif role == "Sheriff" then color = CONFIG.SheriffColor
    elseif role == "Hero" then color = CONFIG.HeroColor
    else color = CONFIG.InnocentColor end
    
    local highlight = Create("Highlight", {
        Name = "OnyyxESP_Highlight",
        Parent = char,
        FillColor = color,
        OutlineColor = Color3.fromRGB(255, 255, 255),
        FillTransparency = 0.7,
        OutlineTransparency = 0.2,
        DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    })
    
    local billboard = Create("BillboardGui", {
        Name = "OnyyxESP_Billboard",
        Parent = hrp,
        AlwaysOnTop = true,
        Size = UDim2.new(0, 120, 0, 50),
        StudsOffset = Vector3.new(0, 3.5, 0)
    })
    
    local frame = Create("Frame", {
        Parent = billboard,
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundColor3 = color,
        BackgroundTransparency = 0.3,
        BorderSizePixel = 0
    })
    AddCorner(frame, UDim.new(0, 8))
    AddStroke(frame, color, 2)
    AddGlow(frame, color, 15)
    
    Create("TextLabel", {
        Parent = frame,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 0.5, 0),
        Font = CONFIG.Font,
        Text = player.Name,
        TextColor3 = Color3.fromRGB(255, 255, 255),
        TextSize = 13
    })
    
    local roleLabel = Create("TextLabel", {
        Parent = frame,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 0, 0.5, 0),
        Size = UDim2.new(1, 0, 0.5, 0),
        Font = CONFIG.FontSmall,
        Text = "🔴 " .. role:upper(),
        TextColor3 = color,
        TextSize = 11
    })
    
    task.spawn(function()
        while billboard and billboard.Parent do
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local dist = (hrp.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                roleLabel.Text = "🔴 " .. role:upper() .. " | " .. math.floor(dist) .. "m"
            end
            task.wait(0.2)
        end
    end)
    
    ESPSystem.ESPObjects[player] = {
        Highlight = highlight,
        Billboard = billboard
    }
end

function ESPSystem.RemoveESP(player)
    if ESPSystem.ESPObjects[player] then
        local objs = ESPSystem.ESPObjects[player]
        if objs.Highlight then objs.Highlight:Destroy() end
        if objs.Billboard then objs.Billboard:Destroy() end
        ESPSystem.ESPObjects[player] = nil
    end
end

function ESPSystem.Update()
    if not ESPSystem.Active then return end
    RoleSystem.DetectRoles()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local role = RoleSystem.CurrentRoles[player] or "Innocent"
            ESPSystem.CreateESP(player, role)
        end
    end
end

--// AUTO SHOOT
local AutoShootSystem = {
    Active = false,
    Connection = nil
}

function AutoShootSystem.Start()
    if AutoShootSystem.Connection then
        AutoShootSystem.Connection:Disconnect()
    end
    
    AutoShootSystem.Connection = RunService.RenderStepped:Connect(function()
        if not AutoShootSystem.Active then return end
        
        local hasGun = false
        local gunTool = nil
        
        if LocalPlayer.Character then
            for _, tool in pairs(LocalPlayer.Character:GetChildren()) do
                if tool:IsA("Tool") and (tool.Name:lower():find("gun") or tool.Name:lower():find("revolver")) then
                    hasGun = true
                    gunTool = tool
                    break
                end
            end
        end
        
        if not hasGun then
            local backpack = LocalPlayer:FindFirstChild("Backpack")
            if backpack then
                for _, tool in pairs(backpack:GetChildren()) do
                    if tool.Name:lower():find("gun") or tool.Name:lower():find("revolver") then
                        tool.Parent = LocalPlayer.Character
                        hasGun = true
                        gunTool = tool
                        break
                    end
                end
            end
        end
        
        if hasGun and gunTool and RoleSystem.Murderer then
            local murdererChar = RoleSystem.Murderer.Character
            if murdererChar then
                local murdererHead = murdererChar:FindFirstChild("Head")
                local murdererHRP = murdererChar:FindFirstChild("HumanoidRootPart")
                
                if murdererHead and murdererHRP then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(murdererHead.Position)
                    
                    if onScreen then
                        local dist = (murdererHead.Position - Camera.CFrame.Position).Magnitude
                        if dist < 200 then
                            Camera.CFrame = CFrame.new(Camera.CFrame.Position, murdererHead.Position)
                            pcall(function()
                                gunTool:Activate()
                            end)
                        end
                    end
                end
            end
        end
    end)
end

function AutoShootSystem.Stop()
    if AutoShootSystem.Connection then
        AutoShootSystem.Connection:Disconnect()
        AutoShootSystem.Connection = nil
    end
end

--// GRAB GUN
local GrabGunSystem = {
    Active = false,
    Connection = nil,
    OriginalPosition = nil
}

function GrabGunSystem.Start()
    GrabGunSystem.Connection = RunService.Heartbeat:Connect(function()
        if not GrabGunSystem.Active then return end
        
        local droppedGun = nil
        for _, obj in pairs(Workspace:GetDescendants()) do
            if obj:IsA("Tool") and obj.Parent == Workspace then
                if obj.Name:lower():find("gun") or obj.Name:lower():find("revolver") then
                    droppedGun = obj
                    break
                end
            end
            if obj:IsA("BasePart") and (obj.Name:lower():find("gun") or obj.Name:lower():find("revolver")) then
                if obj.Parent and obj.Parent:IsA("Model") then
                    droppedGun = obj
                    break
                end
            end
        end
        
        if droppedGun and LocalPlayer.Character then
            local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                if not GrabGunSystem.OriginalPosition then
                    GrabGunSystem.OriginalPosition = hrp.CFrame
                end
                
                local gunPos = droppedGun.Position or droppedGun:GetPivot().Position
                hrp.CFrame = CFrame.new(gunPos + Vector3.new(0, 3, 0))
                task.wait(0.3)
                
                local hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
                if hum then
                    firetouchinterest(hrp, droppedGun, 0)
                    firetouchinterest(hrp, droppedGun, 1)
                end
                
                task.wait(0.2)
                if GrabGunSystem.OriginalPosition then
                    hrp.CFrame = GrabGunSystem.OriginalPosition
                    GrabGunSystem.OriginalPosition = nil
                end
            end
        end
    end)
end

function GrabGunSystem.Stop()
    if GrabGunSystem.Connection then
        GrabGunSystem.Connection:Disconnect()
        GrabGunSystem.Connection = nil
    end
    GrabGunSystem.OriginalPosition = nil
end

--// BHOP SYSTEM (Half-Life 2 Style)
local BhopSystem = {
    Active = false,
    Connection = nil,
    Speed = 50,
    LastJump = 0
}

function BhopSystem.Start()
    BhopSystem.Connection = RunService.Heartbeat:Connect(function()
        if not BhopSystem.Active then return end
        if not LocalPlayer.Character then return end
        
        local hum = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not hum or not hrp then return end
        
        -- Check if on ground
        if hum.FloorMaterial ~= Enum.Material.Air then
            -- Check if space is held (like in HL2)
            if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
                local now = tick()
                if now - BhopSystem.LastJump > 0.08 then
                    BhopSystem.LastJump = now
                    
                    -- Calculate bunny hop direction
                    local cam = Workspace.CurrentCamera
                    local moveDir = hum.MoveDirection
                    
                    if moveDir.Magnitude > 0 then
                        -- Add velocity in movement direction
                        local boost = moveDir.Unit * BhopSystem.Speed
                        hrp.Velocity = Vector3.new(boost.X, hum.JumpPower * 1.2, boost.Z)
                    else
                        hrp.Velocity = Vector3.new(hrp.Velocity.X, hum.JumpPower * 1.2, hrp.Velocity.Z)
                    end
                    
                    -- Auto jump
                    hum:ChangeState(Enum.HumanoidStateType.Jumping)
                end
            end
        else
            -- In air - maintain momentum like HL2
            if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
                local vel = hrp.Velocity
                local horizontalVel = Vector3.new(vel.X, 0, vel.Z)
                
                if horizontalVel.Magnitude < BhopSystem.Speed then
                    local cam = Workspace.CurrentCamera
                    local moveDir = hum.MoveDirection
                    if moveDir.Magnitude > 0 then
                        local airControl = moveDir.Unit * 2
                        hrp.Velocity = vel + Vector3.new(airControl.X, 0, airControl.Z)
                    end
                end
            end
        end
    end)
end

function BhopSystem.Stop()
    if BhopSystem.Connection then
        BhopSystem.Connection:Disconnect()
        BhopSystem.Connection = nil
    end
end

--// TARGET FLING SYSTEM
local TargetFlingSystem = {
    Active = false,
    TargetPlayer = nil,
    SpinSpeed = 50
}

function TargetFlingSystem.Fling(targetName)
    local target = nil
    for _, player in pairs(Players:GetPlayers()) do
        if player.Name:lower():find(targetName:lower()) or 
           player.DisplayName:lower():find(targetName:lower()) then
            target = player
            break
        end
    end
    
    if not target then
        warn("[FLING] Player not found: " .. targetName)
        return false
    end
    
    if not target.Character then
        warn("[FLING] Target has no character")
        return false
    end
    
    local targetHRP = target.Character:FindFirstChild("HumanoidRootPart")
    if not targetHRP then
        warn("[FLING] Target has no HRP")
        return false
    end
    
    TargetFlingSystem.TargetPlayer = target
    
    -- Save our position
    local ourChar = LocalPlayer.Character
    if not ourChar then return false end
    local ourHRP = ourChar:FindFirstChild("HumanoidRootPart")
    if not ourHRP then return false end
    
    local originalPos = ourHRP.CFrame
    
    -- Teleport to target
    ourHRP.CFrame = targetHRP.CFrame * CFrame.new(0, 0, 2)
    task.wait(0.1)
    
    -- Create spin fling
    local spinConnection
    local angle = 0
    local flingTime = tick()
    
    spinConnection = RunService.Heartbeat:Connect(function()
        if tick() - flingTime > 3 then
            spinConnection:Disconnect()
            return
        end
        
        if not target.Character or not target.Character:FindFirstChild("HumanoidRootPart") then
            spinConnection:Disconnect()
            return
        end
        
        if not ourHRP or not ourHRP.Parent then
            spinConnection:Disconnect()
            return
        end
        
        angle = angle + TargetFlingSystem.SpinSpeed * 0.016
        
        -- Spin around target
        local targetPos = target.Character.HumanoidRootPart.Position
        local radius = 3
        local spinPos = targetPos + Vector3.new(
            math.cos(angle) * radius,
            math.sin(angle * 3) * 2 + 3,
            math.sin(angle) * radius
        )
        
        ourHRP.CFrame = CFrame.new(spinPos, targetPos)
        
        -- Push target away (fling)
        local pushDir = (targetPos - spinPos).Unit
        target.Character.HumanoidRootPart.Velocity = pushDir * 500 + Vector3.new(0, 300, 0)
        target.Character.HumanoidRootPart.RotVelocity = Vector3.new(
            math.random(-50, 50),
            math.random(-50, 50),
            math.random(-50, 50)
        )
    end)
    
    -- Wait for fling to finish then return
    task.spawn(function()
        task.wait(3.5)
        
        -- Return to original position
        if ourHRP and ourHRP.Parent then
            ourHRP.CFrame = originalPos
        end
        
        TargetFlingSystem.TargetPlayer = nil
        print("[FLING] Completed fling on " .. target.Name)
    end)
    
    return true
end
